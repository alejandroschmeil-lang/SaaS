        $score = [int]$matches[1]
        [pscustomobject]@{

        }
    }

Get-ChildItem -Recurse -Include *.py -Path app | ForEach-Object {
    $lineCount = (Get-Content $_.FullName).Count
    if ($lineCount -gt 39) {
        [PSCustomObject]@{
            File       = $_.FullName
            LineCount  = $lineCount
        }
    }
} | Sort-Object LineCount -Descending | Format-Table -AutoSize
